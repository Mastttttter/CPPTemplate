            int lp=tree[p].left;
            int rp=tree[p].right;
            int m=l+(r-l)/2;
            tree[lp].lazy=passLazy(tree[lp].lazy,tree[p].lazy),tree[lp].ifLazy=true;
            tree[rp].lazy=passLazy(tree[rp].lazy,tree[p].lazy),tree[rp].ifLazy=true;

        maintain(cl,cr,p);
